Here's a JavaScript program that demonstrates the basic usage of regular expressions:

      // Function to test if a regex pattern matches a string
function testRegex(pattern, str) {
  // Create a regular expression object using the provided pattern
  const regex = new RegExp(pattern);

  // Test the regex pattern against the string
  const result = regex.test(str);

  // Return true if there is a match, false otherwise
  return result;
}

// Test the function with various patterns and strings
console.log(testRegex("hello", "hello world")); // true
console.log(testRegex("hello", "world")); // false

// Test with a pattern that matches any digit
console.log(testRegex("\\d", "123")); // true
console.log(testRegex("\\d", "abc")); // false

// Test with a pattern that matches any word characters followed by "@"
console.log(testRegex("\\w+@", "test@example.com")); // true
console.log(testRegex("\\w+@", "example.com")); // false

Q3

Below is a JavaScript program that demonstrates the use of character classes in regular expressions:

      // Function to search for specific character classes in a given string and return the matches
      function searchCharacterClasses(str) {
        // Define regular expressions for different character classes
        const digitPattern = /\d/g; // Matches any digit
        const uppercasePattern = /[A-Z]/g; // Matches any uppercase letter
        const lowercasePattern = /[a-z]/g; // Matches any lowercase letter
        const specialCharPattern = /[^a-zA-Z0-9]/g; // Matches any special character (non-alphanumeric)
      
        // Store matches for each character class
        const digits = str.match(digitPattern) || [];
        const uppercaseLetters = str.match(uppercasePattern) || [];
        const lowercaseLetters = str.match(lowercasePattern) || [];
        const specialChars = str.match(specialCharPattern) || [];
      
        // Return matches for each character class
        return {
          digits: digits,
          uppercaseLetters: uppercaseLetters,
          lowercaseLetters: lowercaseLetters,
          specialChars: specialChars
        };
      }
      
      // Test the function with a sample string
      const sampleString = "Hello123! How are you?";
      const matches = searchCharacterClasses(sampleString);
      
      // Output the matches for each character class
      console.log("Digits:", matches.digits);
      console.log("Uppercase Letters:", matches.uppercaseLetters);
      console.log("Lowercase Letters:", matches.lowercaseLetters);
      console.log("Special Characters:", matches.specialChars);

Below is a JavaScript program that takes a regex pattern and a string as input. It includes a function that not only checks if there is a match but also extracts specific parts of the matched text using groups:

      // Function to check for a match and extract specific parts using groups
      function matchAndExtract(pattern, str) {
        // Create a regular expression object using the provided pattern
        const regex = new RegExp(pattern);
      
        // Test the regex pattern against the string and extract matches using groups
        const match = regex.exec(str);
      
        // If there is a match
        if (match !== null) {
          // Extract specific parts using groups
          const extractedParts = {
            fullMatch: match[0], // Full matched text
            day: match[1],       // Group for day
            month: match[2],     // Group for month
            year: match[3]       // Group for year
          };
          return extractedParts;
        } else {
          // No match found
          return null;
        }
      }
      
      // Test the function with a pattern to extract parts of a date
      const pattern = /(\d{2})-(\d{2})-(\d{4})/; // Pattern for date in dd-mm-yyyy format
      const inputString = "Today's date is 24-03-2024";
      const result = matchAndExtract(pattern, inputString);
      
      // Output the result
      if (result !== null) {
        console.log("Full Match:", result.fullMatch);
        console.log("Day:", result.day);
        console.log("Month:", result.month);
        console.log("Year:", result.year);
      } else {
        console.log("No match found.");
      }

Below is a JavaScript program that takes the type of package ("standard", "express", or "overnight") as input and uses a switch statement to calculate and print the estimated delivery time based on the package type:

        function estimateDeliveryTime(packageType) {
        let deliveryTime;
      
        switch (packageType) {
          case "standard":
            deliveryTime = "3-5 days";
            break;
          case "express":
            deliveryTime = "1-2 days";
            break;
          case "overnight":
            deliveryTime = "next day";
            break;
          default:
            deliveryTime = "not available";
        }
      
        return deliveryTime;
      }
      
      // Test the function with different package types
      console.log("Package type: standard, Estimated delivery time:", estimateDeliveryTime("standard"));
      console.log("Package type: express, Estimated delivery time:", estimateDeliveryTime("express"));
      console.log("Package type: overnight, Estimated delivery time:", estimateDeliveryTime("overnight"));
      console.log("Package type: unknown, Estimated delivery time:", estimateDeliveryTime("unknown"));
